📄 Apps Script (คัดลอกทั้งหมด) 
// ════════════════════════════════════════════════
//  บันทึกรถ 2ฒฉ 4495 · Google Apps Script v1
//  ชีท: 2ฒฉ4495
// ════════════════════════════════════════════════
const SHEET_NAME = '2ฒฉ4495';

const COLS = {
  date:      'วันที่',
  detail:    'รายละเอียดการใช้รถ',
  requester: 'ผู้ขอเบิก',
  mileOut:   'เลขไมล์ไป',
  timeOut:   'เวลาเบิก',
  mileBack:  'เลขไมล์กลับ',
  timeBack:  'เวลาคืน',
  approver:  'ผู้ให้เบิก',
  note:      'หมายเหตุ',
};

const HDR_ORDER = [
  'วันที่','รายละเอียดการใช้รถ','ผู้ขอเบิก',
  'เลขไมล์ไป','เวลาเบิก',
  'เลขไมล์กลับ','เวลาคืน',
  'ผู้ให้เบิก','หมายเหตุ'
];

function doGet(e) {
  const p = e.parameter;
  const action = p.action || '';
  try {
    if (action === 'ping')      return ok({ pong: true });
    if (action === 'getData')   return ok({ status:'ok', data: getData(p) });
    if (action === 'addRow')    return ok(addRow(p));
    if (action === 'updateRow') return ok(updateRow(p));
    if (action === 'deleteRow') return ok(deleteRow(p));
    if (action === 'getMeta')   return ok(getMeta());
    return ok({ status: 'ready' });
  } catch(err) {
    return ok({ status:'error', message: err.message });
  }
}

function getSheet() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
}

function ensureHeaders(sh) {
  const row1 = sh.getRange(1,1,1,HDR_ORDER.length).getValues()[0];
  if (row1.every(c => c === '' || c === null)) {
    sh.getRange(1,1,1,HDR_ORDER.length).setValues([HDR_ORDER]);
    sh.getRange(1,1,1,HDR_ORDER.length).setFontWeight('bold');
    sh.setFrozenRows(1);
  }
}

function getHeaders(sh) {
  ensureHeaders(sh);
  return sh.getRange(1,1,1,Math.max(sh.getLastColumn(),HDR_ORDER.length))
           .getValues()[0].map(h => String(h).trim());
}

function colIdx(hdrs, name) { return hdrs.indexOf(name); }

function getData(p) {
  const sh = getSheet();
  if (!sh) return [];
  const hdrs = getHeaders(sh);
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return [];
  const vals = sh.getRange(2,1,lastRow-1,hdrs.length).getValues();
  let rows = vals.map((r,i) => {
    const obj = { _row: i+2 };
    hdrs.forEach((h,j) => { obj[h] = r[j] != null ? String(r[j]).trim() : ''; });
    return obj;
  }).filter(r => r['วันที่'] || r['เลขไมล์ไป']);

  if (p.month) {
    rows = rows.filter(r => {
      const d = r['วันที่'] || '';
      // yyyy-mm-dd
      if (d.includes('-')) return d.startsWith(p.month);
      // dd/mm/yyyy หรือ dd/mm/yy
      const parts = d.split('/');
      if (parts.length >= 3) {
        const y = parts[2].length === 2 ? '20' + parts[2] : parts[2].substring(0,4);
        return y + '-' + parts[1].padStart(2,'0') === p.month;
      }
      return false;
    });
  }
  if (p.requester) rows = rows.filter(r => r['ผู้ขอเบิก'] === p.requester);
  return rows;
}

function addRow(p) {
  const sh = getSheet();
  if (!sh) return { ok:false, error:'ไม่พบ Sheet' };
  const hdrs = getHeaders(sh);
  const newRow = Array(hdrs.length).fill('');
  HDR_ORDER.forEach(col => {
    const idx = colIdx(hdrs, col);
    if (idx >= 0 && p[col] !== undefined) newRow[idx] = p[col];
  });
  sh.appendRow(newRow);
  return { ok:true, row: sh.getLastRow() };
}

function updateRow(p) {
  const sh = getSheet();
  if (!sh) return { ok:false, error:'ไม่พบ Sheet' };
  const hdrs = getHeaders(sh);
  const rowNum = parseInt(p._row);
  if (!rowNum || rowNum < 2) return { ok:false, error:'row ไม่ถูกต้อง' };
  const existing = sh.getRange(rowNum,1,1,hdrs.length).getValues()[0];
  HDR_ORDER.forEach(col => {
    const idx = colIdx(hdrs, col);
    if (idx >= 0 && p[col] !== undefined && p[col] !== '') {
      existing[idx] = p[col];
    }
  });
  sh.getRange(rowNum,1,1,hdrs.length).setValues([existing]);
  return { ok:true };
}

function deleteRow(p) {
  const sh = getSheet();
  if (!sh) return { ok:false, error:'ไม่พบ Sheet' };
  const rowNum = parseInt(p._row);
  if (!rowNum || rowNum < 2) return { ok:false, error:'row ไม่ถูกต้อง' };
  sh.deleteRow(rowNum);
  return { ok:true };
}

function getMeta() {
  const sh = getSheet();
  if (!sh) return { requesters:[], approvers:[], details:[] };
  const hdrs = getHeaders(sh);
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return { requesters:[], approvers:[], details:[] };
  const vals = sh.getRange(2,1,lastRow-1,hdrs.length).getValues();
  const requesters = new Set(), approvers = new Set(), details = new Set();
  const ri = colIdx(hdrs,'ผู้ขอเบิก');
  const ai = colIdx(hdrs,'ผู้ให้เบิก');
  const di = colIdx(hdrs,'รายละเอียดการใช้รถ');
  vals.forEach(r => {
    if (ri>=0 && r[ri]) requesters.add(String(r[ri]).trim());
    if (ai>=0 && r[ai]) approvers.add(String(r[ai]).trim());
    if (di>=0 && r[di]) details.add(String(r[di]).trim());
  });
  return {
    requesters: [...requesters].filter(Boolean),
    approvers:  [...approvers].filter(Boolean),
    details:    [...details].filter(Boolean),
  };
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
